1 /** @namespace */
  2 app = {};
  3 /**
  4 * Takes an iterable and transforms it into an array.
  5 */
  6 app.toArray = function (iterable) {
  7 		return Array.prototype.slice.call(iterable);
  8 	};
  9 /**
 10 * Inspired by Cobra.Class.method(callable, self)
 11 * It will wrap a function to give it a new _this_ context.
 12 */
 13 app.methodize = function (/** the function to wrap inside a new function call to give it a new this context.*/ callable, /** self is the python verb for _this_ class.*/ self) {
 14 	
 15 	function method () {
 16 		var args = app.toArray(arguments);
 17 		return callable.apply(self, args);
 18 	}
 19 	method.callable = callable;
 20 	return method;
 21 };
 22 /**
 23 * This function takes an object and wraps each of it's functions in a call that makes sure that 
 24 * this will be bound to the object.
 25 */
 26 app.protectThis = function(object){
 27 	for (key in object) {
 28 		if (object.hasOwnProperty(key)) {
 29 			var member = object[key];
 30 	
 31 			if (typeof member == 'function') {
 32 				object[key] = app.methodize(member, object);
 33 			}
 34 		}
 35 	}
 36 	return object;
 37 };
 38 /**
 39 * This function takes an object, a name to the function to override and a new function that overrides the old.
 40 * In order to pass on execution to the old function you use this(args) inside the new function. 
 41 */
 42 app.override = function(self, /**The name of the function to override.*/ key, /**The child function overriding the parent.*/ child){
 43     var parent = self[key];
 44     /** @private */
 45     self[key] = function(){
 46         return child.apply(parent,arguments);
 47     };
 48 };
